\section*{Self-Guided Derivation & Companion Tasks (Active)}

\begin{enumerate}
    \item \textbf{[D1]} Canonical Quantization of O1:  
    Derive Hamiltonian structure from \(\pi^\mu(\lambda)\), \(\xi^\mu(\lambda)\). Show full canonical algebra.  
    \textit{Output:} \texttt{SAT\_D1\_CANONICAL\_QUANTIZATION.txt}

    \item \textbf{[D2]} Field Mapping for \(\psi(x)\):  
    Define field-level expressions for fermionic bundles. Clarify \(\tau(x)\) dependence and \( \theta_4(x) \)-related mass.  
    \textit{Output:} \texttt{SAT\_D2\_FERMION\_FIELD\_MAPPING.txt}

    \item \textbf{[D3]} Topological Mass Quantization:  
    Derive scaling behavior of \( Q \) in terms of knot theory invariants. Correlate predicted mass steps to particle families.  
    \textit{Output:} \texttt{SAT\_D3\_TOPOLOGICAL\_SUPPRESSION.txt}

    \item \textbf{[D4]} Strain Tensor and Curvature:  
    Derive Ricci curvature from strain field \( S_{\mu\nu} \). Match Einstein-Hilbert form from emergent foliation stress.  
    \textit{Output:} \texttt{SAT\_D4\_STRAIN\_CURVATURE\_MAPPING.txt}

    \item \textbf{[D5]} Gauge Curvature via Linking:  
    Show how local linking density translates into gauge curvature tensors \( F_{\mu\nu} \). Connect to SU(2), SU(3) bundle class.  
    \textit{Output:} \texttt{SAT\_D5\_GAUGE\_TOPOLOGY.txt}

    \item \textbf{[D6]} Translation of Standard Terms:  
    Map classical field actions (e.g., Maxwell, Dirac, Yang-Mills) into SAT-structured analogues. Identify translation frames.  
    \textit{Output:} \texttt{SAT\_D6\_LEGACY\_TRANSLATION.txt}
\end{enumerate}
