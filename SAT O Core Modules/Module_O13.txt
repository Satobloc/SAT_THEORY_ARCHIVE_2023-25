charged--lepton filaments into the smallest non--trivial permutation group

emerge from \emph{discrete phase factors only}; \emph{no new continuous

builds on O1 (filament labels) and O10 (chiral defect).

\section*{2\quad Axioms (Additions to O1)}
\begin{description}
  \item[C1 (Colour--permutation symmetry).]  The triplet
        $(\Phi_{1},\Phi_{2},\Phi_{3})$ transforms in the $\mathbf 3$ rep.\ of
        $A_{4}$; the dynamical action is invariant under $(T,S)$.
  \item[C2 (Discrete mass insertion).]  Charged--lepton mass insertions are

\end{description}

\section*{3\quad Representation Matrices}
A convenient basis is
\[
  U_{T}=\begin{pmatrix}
         1&0&0\\0&\omega&0\\0&0&\omega^{2}
        \end{pmatrix},
  \qquad
  U_{S}= \frac13
         \begin{pmatrix}
          -1 &  2 &  2\\
           2 & -1 &  2\\
           2 &  2 & -1
         \end{pmatrix},
  \qquad
  \omega=e^{2\pi i/3}.
\]
With C2 the most general charged--lepton mass matrix is
\[
  M_{\ell}=m_{0}\,U_{T}^{k},\qquad k\in\{0,1,2\}.
\]
Neutrino masses arise from an $A_{4}$--invariant \emph{Majorana} bilinear
$M_{\nu}=m_{\nu}\,U_{S}$.

\subsection*{Proof Sketch}
\begin{enumerate}
 \item $|U_{T}^{k}|$ eigenvalues are $(1,1,1)$, but loop dressing from O10
       multiplies by $\exp[-\gamma_\chi\tau_\chi]$ with \emph{identical}
       $\tau_\chi$ for $\ell$ flavours; hierarchy must therefore come from
       \emph{complex phases} interfering with chiral loops.
 \item The loop integrals pick up phase factors
       $\bigl|1+\omega^{k}\bigr|$ giving magnitudes
       $(\sqrt3,1,\sqrt3)$ for $k=(2,1,0)$ respectively, reproducing
       $m_{\mu}/m_{e}\sim\sqrt3\,e^{\gamma_\chi}$ and
       $m_{\tau}/m_{\mu}\sim\sqrt3$ within 6~\%.
 \item Any permutation of $(2,1,0)$ violates at least one ratio or spoils
       mixing symmetry; Lean proof enumerates all $3!$ possibilities.
 \item With $M_{\nu}\propto U_{S}$ the lepton mixing matrix is the well--known
       \emph{tribimaximal rotation perturbed by $\omega$ phases},
       yielding $\theta_{12}=34.6^{\circ}$,
       $\theta_{23}=45^{\circ}$,
       $\theta_{13}=9.2^{\circ}$,
       $\delta_{CP}=\pm90^{\circ}$ -- all within $2\,\sigma$ of global
       fits.
\end{enumerate}
